85static float read_raw_current(current_sensor_t sensor) {
86 int adc_raw = 0;
87 int voltage = 0; // Use `int` instead of `uint32_t` as required by `adc_cali_raw_to_voltage`
88
89 if (sensor == SENSOR_1) {
90 adc_oneshot_read(adc_handle, SENSOR_1_CHANNEL, &adc_raw);
91 } else if (sensor == SENSOR_2) {
92 adc_oneshot_read(adc_handle, SENSOR_2_CHANNEL, &adc_raw);
93 }
94
95 if (cali_handle) {
96 adc_cali_raw_to_voltage(cali_handle, adc_raw, &voltage); // Calibrated voltage in mV
97 } else {
98 ESP_LOGW(TAG, "ADC calibration not available. Using raw value.");
99 voltage = adc_raw; // Use raw ADC reading if calibration is unavailable
100 }
101
102 float current = ((float)voltage / 1000.0) / (CURRENT_SENSE_RESISTOR * CURRENT_SENSE_GAIN); // Current in amperes
103 return current;
104}

114float current_sensing_get_debounced(current_sensor_t sensor) {
115 float sum = 0;
116 int index = (sensor == SENSOR_1) ? 0 : 1;
117
118 for (int i = 0; i < 5; i++) {
119 sum += last_measurements[index][i];
120 }
121 return sum / 5.0;

131void current_sensing_update(current_sensor_t sensor) {
132 float current = read_raw_current(sensor);
133 int index = (sensor == SENSOR_1) ? 0 : 1;
134
135 last_measurements[index][measurement_index[index]] = current;
136 measurement_index[index] = (measurement_index[index] + 1) % 5;
137
138 ESP_LOGI(TAG, "Updated current measurement buffer for Sensor %d. Raw Current: %.3f A", sensor + 1, current);
139}
