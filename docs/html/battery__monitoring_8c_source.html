44void battery_monitoring_init() {
45 ESP_LOGI(TAG, "Initializing battery monitoring module...");
46
47 // ADC oneshot configuration
48 adc_oneshot_unit_init_cfg_t adc_config = {
49 .unit_id = ADC_UNIT_1

52
53 // Configure channel

56 .atten = ADC_ATTEN_DB_12

58 adc_oneshot_config_channel(adc_handle, BATTERY_ADC_CHANNEL, &channel_config);

60 // ADC calibration configuration
61 adc_cali_line_fitting_config_t cali_config = {
62 .unit_id = ADC_UNIT_1,
63 .atten = ADC_ATTEN_DB_12,
64 .bitwidth = ADC_BITWIDTH_DEFAULT
65 };
66 if (adc_cali_create_scheme_line_fitting(&cali_config, &cali_handle) != ESP_OK) {
67 ESP_LOGE(TAG, "Failed to initialize ADC calibration!");
68 cali_handle = NULL;
69 }
70
71 // Initialize debounce buffer
72 for (int i = 0; i < DEBOUNCE_SAMPLES; i++) {
73 last_measurements[i] = 0.0;
74 }

76 ESP_LOGI(TAG, "Battery monitoring module initialized successfully.");

86static float read_raw_voltage() {
87 int adc_raw = 0;
88 int voltage = 0;
89
90 adc_oneshot_read(adc_handle, BATTERY_ADC_CHANNEL, &adc_raw);
91
92 if (cali_handle) {
93 adc_cali_raw_to_voltage(cali_handle, adc_raw, &voltage); // Calibrated voltage in mV
94 } else {
95 ESP_LOGW(TAG, "ADC calibration not available. Using raw value.");
96 voltage = adc_raw; // Use raw ADC reading if calibration is unavailable
97 }
98
99 return ((float)voltage / 1000.0) * VOLTAGE_DIVIDER_RATIO * 3.0; // Scale to series battery voltage
100}

109static float get_debounced_voltage() {
110 float sum = 0.0;
111 for (int i = 0; i < DEBOUNCE_SAMPLES; i++) {
112 sum += last_measurements[i];
113 }
114 return sum / DEBOUNCE_SAMPLES;
115}

120static void update_debounce_buffer(float voltage) {
121 last_measurements[measurement_index] = voltage;
122 measurement_index = (measurement_index + 1) % DEBOUNCE_SAMPLES;
123}

133static int calculate_soc(float voltage) {
134 if (voltage >= BATTERY_FULL_VOLTAGE) {
135 return 100;
136 } else if (voltage <= BATTERY_EMPTY_VOLTAGE) {
137 return 0;
138 } else {
139 return (int)(((voltage - BATTERY_EMPTY_VOLTAGE) / (BATTERY_FULL_VOLTAGE - BATTERY_EMPTY_VOLTAGE)) * 100);
140 }
141}

148void battery_monitoring_run() {
149 float raw_voltage = read_raw_voltage();
150 update_debounce_buffer(raw_voltage);
151
152 float debounced_voltage = get_debounced_voltage();
153 int soc = calculate_soc(debounced_voltage);
154
155 ESP_LOGI(TAG, "Battery Voltage (Raw): %.2fV", raw_voltage);
156 ESP_LOGI(TAG, "Battery Voltage (Debounced): %.2fV", debounced_voltage);
157 ESP_LOGI(TAG, "State of Charge: %d%%", soc);
158
159 if (debounced_voltage <= BATTERY_EMPTY_VOLTAGE) {
160 ESP_LOGW(TAG, "Battery voltage critical! Please recharge.");
161 }
162}
