24void nvm_init() {

26 if (err == ESP_ERR_NVS_NO_FREE_PAGES || err == ESP_ERR_NVS_NEW_VERSION_FOUND) {
27 ESP_LOGW(TAG, "Erasing and reinitializing NVS due to errors...");
